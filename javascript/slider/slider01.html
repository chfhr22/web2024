window.onload = function(){
        let currentIndex = 0;       // 현재 이미지
        const slider = document.querySelectorAll(".slider");        // 모든 이미지를 변수에 저장

        // 3초에 한번씩 이미지를 변경
        setInterval(() => {
            let nextInderx = (currentIndex + 1) % slider.length;

            slider[currentIndex].style.opacity = "0";                   // 첫번째 이미지가 사라짐
            slider[nextInderx].style.opacity = "1";                     // 두번째 이미지가 나타남 
            slider.forEach(img => img.style.transition = "all 1s");     // 이미지 애니메이션 추가
            currentIndex = nextInderx;                                  // 두번째 인덱스값을 현재 인덱스값에 저장
        }, 3000);
}
$(function(){
        let currentIndex = 0;       // 현재 이미지

        setInterval(function(){
            let nextIndex = (currentIndex + 1) % 5;
            
            $(".slider").eq(currentIndex).fadeOut(1200);    // 첫번째 이미지 사라지기 (fadeOut)
            $(".slider").eq(nextIndex).fadeIn(1200);        // 두번째 이미지 나타나기 (fadeIn)

            currentIndex = nextIndex;
        }, 3000);
});
